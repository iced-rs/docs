Creates a keyed Column with the given children.

Keyed columns distribute content vertically while keeping continuity.

ยงExample

use iced::widget::keyed_column;

enum Message {
    // ...
}

fn view(state: &State) -> Element<'_, Message> {
    keyed_column![
        (0, "Item 0"),
        (1, "Item 1"),
        (2, "Item 2"),
    ].into()
}
